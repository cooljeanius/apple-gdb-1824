42/* This header declares most of the API for dealing with minimal
43 symbols and minimal symbol tables. A few things are declared
44 elsewhere; see below.
45
46 A minimal symbol is a symbol for which there is no direct debug
47 information. For example, for an ELF binary, minimal symbols are
48 created from the ELF symbol table.
49
50 For the definition of the minimal symbol structure, see struct
51 minimal_symbol in symtab.h.
52
53 Minimal symbols are stored in tables attached to an objfile; see
54 objfiles.h for details. Code should generally treat these tables
55 as opaque and use functions provided by minsyms.c to inspect them.
56*/

75/* Record a new minimal symbol. This is the "full" entry point;
76 simpler convenience entry points are also provided below.
77
78 This returns a new minimal symbol. It is ok to modify the returned
79 minimal symbol (though generally not necessary). It is not ok,
80 though, to stash the pointer anywhere; as minimal symbols may be
81 moved after creation. The memory for the returned minimal symbol
82 is still owned by the minsyms.c code, and should not be freed.
83
84 Arguments are:
85
86 NAME - the symbol's name
87 NAME_LEN - the length of the name
88 COPY_NAME - if true, the minsym code must make a copy of NAME. If
89 false, then NAME must be NUL-terminated, and must have a lifetime
90 that is at least as long as OBJFILE's lifetime.
91 ADDRESS - the address of the symbol
92 MS_TYPE - the type of the symbol
93 SECTION - the symbol's section
94 appropriate obj_section for the minimal symbol. This can be NULL.
95 OBJFILE - the objfile associated with the minimal symbol. */
96
97struct minimal_symbol *prim_record_minimal_symbol_full
98 (const char *name,
99 int name_len,
100 int copy_name,
101 CORE_ADDR address,
102 enum minimal_symbol_type ms_type,
103 int section,
104 struct objfile *objfile) ATTRIBUTE_W_U_R;
